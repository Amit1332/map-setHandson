we make a function countAlpha and pass argument when call this function and take str as a parameter inside a function make an Object using Map() Method the run a loop to get a char of str and check the condition if char is exist in Map Object name alphabetMap variable then set that char and get that char presence else not in object then only set that char , to print the given format output firstly we have to get key of object and the count that keys so i get allkeys as an array then run a loop on array and print keys and their count 
let a = "abbbcdeeefff"
function countAlpha(str){
let alphabetMap =new Map();

for(let char of str){
if(alphabetMap.has(char)){
 alphabetMap.set(char,
 alphabetMap.get(char)+1)
}else{
 alphabetMap.set(char,1)
 }
 }
 const allkeys = 
 Array.from(alphabetMap.keys())
for(let key of allkeys){
  console.log(`${key}=
  ${alphabetMap.get(key)}`)
    }
 }
